	picture = GFX_event_equestria_fail
	immediate = {log = "[GetDateText]: [Root.GetName]: event lunarep.10"}
	is_triggered_only = yes
	option = {
		name = lunarep.10.a
		ai_chance = {
			factor = 3
			modifier = {
				factor = 0
				any_country = {
					OR = {
						is_justifying_wargoal_against = ROOT
						has_wargoal_against = ROOT
						has_war_with = ROOT
					}
				}
			}
		}
		send_equipment = {
		    equipment = infantry_equipment
		    amount = 1000
		    target = NLR
		}
		add_manpower = -3000
		NLR = { 
			add_manpower = 3000
			set_country_flag = lunarep_10_yes 
			add_to_variable = { var = NLR_number_of_sent_guns value = 1000 }
			add_to_variable = { var = NLR_number_of_sent_manpower value = 3000 }
		}
	}
	option = {
		name = lunarep.10.b
		ai_chance = {
			factor = 3
			modifier = {
				NOT = { is_pony_race = yes }
				factor = 2
			}
			modifier = {
				NOT = {
					is_neighbor_of = NLR
					is_neighbor_of = EQR
					is_neighbor_of = event_target:solar_empire
				}
				factor = 1.5
			}
			modifier = {
				NOT = {
					has_government = democratic 
				}
				factor = 3
			}
			modifier = {
				original_tag = CRY
				factor = 0.1
			}
			modifier = {
				original_tag = EQC
				factor = 0.1
			}
		}
	}
}
