		context_type = selected_state_context

		}

					AND = {

							var = daybreaker_magic_reserves 
							value = sol_heatwave_cost
							compare = greater_than_or_equals
						}
					}
					AND = {
						check_variable = { sol_magic_currently_selected_spell = 4 }
					}
				}
			}
			THIS = {
				NOT = {
					AND = {
						has_state_flag = SOL_state_has_spell_heatwave
						ROOT = { check_variable = { sol_magic_currently_selected_spell = 1 } }
					}
				}
			}
		}
		
		window_name = "SOL_wrath_of_the_sun_button_clickable_window"
		
		effects = {
			SOL_wrath_of_the_sun_button_click = {
				hidden_effect = {
					if = {
						limit = {
							ROOT = { check_variable = { sol_magic_currently_selected_spell = 1 } }
						}
						THIS = {
							set_state_flag = SOL_state_has_spell_heatwave
							add_province_modifier = {
								static_modifiers = {
									SOL_heatwave_effect
								}
								province = {
									all_provinces = yes
								}
							}
							add_dynamic_modifier = { modifier = SOL_dynamic_heatwave_effect }
						}
						EQS = {
							subtract_from_variable = { var = daybreaker_magic_reserves value = sol_heatwave_cost }
							add_to_variable = { var = sol_heatwave_number value = 1 }
							SOL_calculate_magic = yes
						}
						sound_effect = "sol_fire_ev"
					}
					if = {
						limit = {
							ROOT = { check_variable = { sol_magic_currently_selected_spell = 4 } }
						}
						EQS = {
							launch_nuke = {
								state = PREV
								use_nuke = no
							}
						}
					}
				}
			}
		}
	}
